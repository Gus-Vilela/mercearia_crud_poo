CREATE TABLE PRODUTOS(
CODPRODUTO int unsigned NOT NULL AUTO_INCREMENT,
DESC_PRODUTO VARCHAR(255),
PRECO DOUBLE NOT NULL,
IMAGEM VARCHAR(255),
UNIDADE_MEDIDA VARCHAR(50) NOT NULL,
PRIMARY KEY(CODPRODUTO)
);
CREATE TABLE VENDEDOR(
CODVENDEDOR int unsigned NOT NULL AUTO_INCREMENT,
NOME VARCHAR(255) NOT NULL,
SALARIOBASE DOUBLE NOT NULL,
PERCCOMISSAO DOUBLE NOT NULL,
PRIMARY KEY(CODVENDEDOR)
);
CREATE TABLE VENDA(
CODVENDA int unsigned NOT NULL AUTO_INCREMENT,
FORMAPAGTO SMALLINT UNSIGNED NOT NULL,
DATAVENDA DATE NOT NULL,
CODVENDEDOR int unsigned NOT NULL,
PRIMARY KEY(CODVENDA),
FOREIGN KEY(CODVENDEDOR) REFERENCES VENDEDOR(CODVENDEDOR)
);
CREATE TABLE PRODUTOSVENDIDOS(
CODPRODUTO int unsigned NOT NULL,
CODVENDA int unsigned NOT NULL,
QUANTIDADE int unsigned NOT NULL,
PRIMARY KEY (CODPRODUTO, CODVENDA),
FOREIGN KEY (CODPRODUTO) REFERENCES PRODUTOS(CODPRODUTO),
FOREIGN KEY (CODVENDA) REFERENCES VENDA(CODVENDA)
);
