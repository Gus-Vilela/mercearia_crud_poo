CREATE TABLE PRODUTOS(
CODPRODUTO INT AUTO_INCREMENT,
DESC_PRODUTO VARCHAR(100),
PRECO DOUBLE NOT NULL,
IMAGEM VARCHAR(250),
UNIDADE_MEDIDA VARCHAR(50) NOT NULL,
PRIMARY KEY(CODPRODUTO)
);
CREATE TABLE VENDEDOR(
CODVENDEDOR INT AUTO_INCREMENT,
NOME VARCHAR(100),
SALARIOBASE DOUBLE NOT NULL,
PERCCOMISSAO DOUBLE NOT NULL,
PRIMARY KEY(CODVENDEDOR)
);
CREATE TABLE PAGAMENTO(
CODPAGAMENTO INT AUTO_INCREMENT,
PRIMARY KEY(CODPAGAMENTO),
FORMAPGTO ENUM('CARTAO DE CREDITO','DINHEIRO', 'BOLETO','DEPOSITO','CONVENIO')
);
CREATE TABLE VENDADIARIA(
CODVENDA INT AUTO_INCREMENT,
VALORTOTAL DOUBLE,
PRIMARY KEY(CODVENDA),
CODPRODUTO INT NOT NULL,
CODPAGAMENTO INT NOT NULL,
CODVENDEDOR INT NOT NULL,
FOREIGN KEY(CODVENDEDOR) REFERENCES VENDEDOR(CODVENDEDOR),
FOREIGN KEY(CODPAGAMENTO) REFERENCES PAGAMENTO(CODPAGAMENTO)
);
CREATE TABLE PRODUTOSVENDIDOS(
CODVENDA INT NOT NULL,
CODPRODUTO INT NOT NULL,
QTDVENDIDA INT NOT NULL,
PRECO DOUBLE NOT NULL,
PRIMARY KEY(CODVENDA, CODPRODUTO),
FOREIGN KEY(CODVENDA) REFERENCES VENDADIARIA(CODVENDA),
FOREIGN KEY(CODPRODUTO) REFERENCES PRODUTOS(CODPRODUTO)
);
